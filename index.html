//////////////////////////////////////////////////////// // SETUP //////////////////////////////////////////////////////// const canvas = document.getElementById("game") const scene = new THREE.Scene() scene.background = new THREE.Color(0x87ceeb) const camera = new THREE.PerspectiveCamera( 75, window.innerWidth/window.innerHeight, 0.1, 1000 ) const renderer = new THREE.WebGLRenderer({ canvas:canvas, antialias:true }) renderer.setSize(window.innerWidth,window.innerHeight) //////////////////////////////////////////////////////// // LUCI //////////////////////////////////////////////////////// const ambient = new THREE.AmbientLight(0xffffff,0.6) scene.add(ambient) const sun = new THREE.DirectionalLight(0xffffff,1) sun.position.set(50,100,50) scene.add(sun) //////////////////////////////////////////////////////// // TERRENO //////////////////////////////////////////////////////// const ground = new THREE.Mesh( new THREE.PlaneGeometry(500,500), new THREE.MeshLambertMaterial({color:0x2ecc71}) ) ground.rotation.x=-Math.PI/2 scene.add(ground) //////////////////////////////////////////////////////// // PLAYER //////////////////////////////////////////////////////// const player = new THREE.Mesh( new THREE.BoxGeometry(2,4,2), new THREE.MeshLambertMaterial({color:0x3498db}) ) player.position.y=2 scene.add(player) //////////////////////////////////////////////////////// // CAMERA //////////////////////////////////////////////////////// camera.position.set(0,6,10) //////////////////////////////////////////////////////// // MOVIMENTO //////////////////////////////////////////////////////// const move={ up:false, down:false, left:false, right:false } function bind(id,dir){ const b=document.getElementById(id) b.addEventListener("pointerdown",()=>move[dir]=true) b.addEventListener("pointerup",()=>move[dir]=false) b.addEventListener("pointerleave",()=>move[dir]=false) } bind("up","up") bind("down","down") bind("left","left") bind("right","right") //////////////////////////////////////////////////////// // SALTO //////////////////////////////////////////////////////// let velocityY=0 let onGround=true document.getElementById("jump").onclick=()=>{ if(onGround){ velocityY=0.25 onGround=false } } //////////////////////////////////////////////////////// // SPARO //////////////////////////////////////////////////////// const bullets=[] document.getElementById("shoot").onclick=()=>{ const bullet=new THREE.Mesh( new THREE.SphereGeometry(0.3), new THREE.MeshBasicMaterial({color:0xffff00}) ) bullet.position.copy(player.position) bullet.velocity=new THREE.Vector3(0,0,-1) bullets.push(bullet) scene.add(bullet) } //////////////////////////////////////////////////////// // COSTRUZIONI //////////////////////////////////////////////////////// document.getElementById("build").onclick=()=>{ const wall=new THREE.Mesh( new THREE.BoxGeometry(5,5,1), new THREE.MeshLambertMaterial({color:0x8e5a2b}) ) wall.position.set( player.position.x, 2, player.position.z-5 ) scene.add(wall) } //////////////////////////////////////////////////////// // BOT //////////////////////////////////////////////////////// const bots=[] function spawnBot(){ const bot=new THREE.Mesh( new THREE.BoxGeometry(2,4,2), new THREE.MeshLambertMaterial({color:0xe74c3c}) ) bot.position.set( Math.random()*100-50, 2, Math.random()*100-50 ) bots.push(bot) scene.add(bot) } for(let i=0;i<10;i++) spawnBot() //////////////////////////////////////////////////////// // STORM //////////////////////////////////////////////////////// let stormRadius=200 function updateStorm(){ stormRadius-=0.02 document.getElementById("storm").innerText="Storm:"+Math.floor(stormRadius) const dist=Math.sqrt( player.position.x*player.position.x+ player.position.z*player.position.z ) if(dist>stormRadius){ playerHP-=0.1 } } //////////////////////////////////////////////////////// // STATS //////////////////////////////////////////////////////// let playerHP=100 //////////////////////////////////////////////////////// // LOOP //////////////////////////////////////////////////////// function animate(){ requestAnimationFrame(animate) if(move.up) player.position.z-=0.3 if(move.down) player.position.z+=0.3 if(move.left) player.position.x-=0.3 if(move.right) player.position.x+=0.3 velocityY-=0.01 player.position.y+=velocityY if(player.position.y<=2){ player.position.y=2 velocityY=0 onGround=true } bullets.forEach(b=>{ b.position.add(b.velocity) }) bots.forEach(bot=>{ const dir=new THREE.Vector3() dir.subVectors(player.position,bot.position) const dist=dir.length() dir.normalize() bot.position.add(dir.multiplyScalar(0.03)) if(dist<8){ playerHP-=0.05 } }) updateStorm() camera.position.x=player.position.x camera.position.z=player.position.z+10 camera.lookAt(player.position) document.getElementById("hp").innerText="HP:"+Math.floor(playerHP) document.getElementById("players").innerText="Enemies:"+bots.length renderer.render(scene,camera) } animate() //////////////////////////////////////////////////////// // RESIZE //////////////////////////////////////////////////////// window.addEventListener("resize",()=>{ camera.aspect=window.innerWidth/window.innerHeight camera.updateProjectionMatrix() renderer.setSize(window.innerWidth,window.innerHeight) })
